import qrcode # Definindo o link e o texto link = "https://pagina-relatorio.vercel.app" texto_adicional = "Acesse o relatório aqui!" # Combinando o texto e o link conteudo = f"{texto_adicional}\n{link}" # Gerando o QR Code qr = qrcode.QRCode( version=1, error_correction=qrcode.constants.ERROR_CORRECT_L, box_size=10, border=4, ) # Adicionando o conteúdo ao QR Code qr.add_data(conteudo) qr.make(fit=True) # Criando a imagem do QR Code img = qr.make_image(fill='black', back_color='white') # Salvando o QR Code em um arquivo img.save("qr_code_com_texto_e_link.png") print("QR Code gerado com sucesso!")
