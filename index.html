import qrcode # Texto que você quer incluir no QR Code texto = "Acesse o relatório aqui: https://pagina-relatorio.vercel.app" # Gerando o QR Code qr = qrcode.QRCode( version=1, error_correction=qrcode.constants.ERROR_CORRECT_L, box_size=10, border=4, ) # Adicionando o texto ao QR Code qr.add_data(texto) qr.make(fit=True) # Criando a imagem do QR Code img = qr.make_image(fill='black', back_color='white') # Salvando o QR Code em um arquivo img.save("qr_code_com_texto.png") print("QR Code gerado com sucesso!")
